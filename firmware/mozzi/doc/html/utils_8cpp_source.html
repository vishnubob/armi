00078 Q16n16  Q16n16_mtof(Q16n16 midival_fractional)
00079 {
00080                 Q16n16 diff_fraction;
00081                 unsigned char index = midival_fractional >> 16;
00082                 unsigned int fraction = (unsigned int) midival_fractional; // keeps low word
00083                 Q16n16 freq1 = (Q16n16) pgm_read_dword(midiToFreq + index);
00084                 Q16n16 freq2 = (Q16n16) pgm_read_dword(midiToFreq + (index+1));
00085                 Q16n16 difference = freq2 - freq1;
00086                 if (difference>=65536)
00087                 {
00088                                 diff_fraction = ((difference>>8) * fraction) >> 8;
00089                 }
00090                 else
00091                 {
00092                                 diff_fraction = (difference * fraction) >> 16;
00093                 }
00094                 return (Q16n16) (freq1+ diff_fraction);

00148 unsigned long xorshift96()

00348 int receiveAnalogRead()
00349 {
00350                 unsigned char low, high;
00351 #if defined(ADCSRA) && defined(ADCL)
00352                 // ADSC is cleared when the conversion finishes
00353                 while (bit_is_set(ADCSRA, ADSC))
00354                                 ;
00355 
00356                 // we have to read ADCL first; doing so locks both ADCL
00357                 // and ADCH until ADCH is read.  reading ADCL second would
00358                 // cause the results of each conversion to be discarded,
00359                 // as ADCL and ADCH would be locked when it completed.
00360                 low  = ADCL;
00361                 high = ADCH;
00362 #else
00363                 // we dont have an ADC, return 0
00364                 low  = 0;
00365                 high = 0;
00366 #endif
00367 
00368                 // combine the two bytes
00369                 return (high << 8) | low;
00370 }
